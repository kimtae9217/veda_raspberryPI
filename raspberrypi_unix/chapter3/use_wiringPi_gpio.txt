# fetch the source
sudo apt install git
git clone https://github.com/WiringPi/WiringPi.git
cd WiringPi

# build the package
./build debian
mv debian-template/wiringpi_3.10_arm64.deb .

# install it
sudo dpkg -i ./wiringpi_3.10_arm64.deb
